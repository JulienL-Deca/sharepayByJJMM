-- afficher les dépenses d'un évènement
  SELECT * FROM expenses INNER JOIN events ON (expenses.event_id = events.id) WHERE events.id = '354e3ffb-da5b-41e8-a1cc-5d14ab91c4fd'; --events.id == réunion équipe

  -- afficher les dépenses d'un évènement avec le nom des payeurs (et pas le uuid)
  SELECT events.name, expenses.description, expenses.date, expenses.amount, participants.nickname
  FROM expenses
  INNER JOIN events ON (expenses.event_id = events.id)
  INNER JOIN participants ON (participants.id = expenses.paid_by)
  WHERE events.id = '354e3ffb-da5b-41e8-a1cc-5d14ab91c4fd';--events.id == réunion équipe

-- afficher toutes les dépenses par paid_by et receveur
  SELECT expenses.date, expenses.amount, expenses.description, participants.nickname, p.nickname AS paid_by_name
  FROM expenses
  INNER JOIN expense_participants ON (expense_participants.expense_id = expenses.id)
  INNER JOIN participants ON (participants.id = expense_participants.participant_id)
  INNER JOIN participants p ON (p.id = expenses.paid_by)
  WHERE expenses.id = '4a895f5c-d376-4d75-9d1c-46865a44924a';
